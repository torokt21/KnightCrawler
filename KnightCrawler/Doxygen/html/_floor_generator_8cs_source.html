5 /* ===============================
6  * Idea:
7  * Floor difficulty translates to the number of rooms between the starting room and the exit room. Higher difficulty = more rooms.
8  * This number is split into two numbers (positive and negative). Horizontal (h) and vertical (v) offset. These numbers mean which
9  * direction the player needs to go in order get to the exit the fastest.
10  * ----------
11  * Example:
12  * v = 5
13  * h = -2
14  * would mean that the exit room is 5 rooms down, and 2 rooms to the left from the starting room.
15  * ----------
16  * After we know where he two rooms are - compared to each other, we need to connect them. This is done by randomizing the order of
17  * the directions the rooms follow eachother.
18  * ----------
19  * Based on the previous example we could go in this order
20  * up up up up up left left
21  * But by randomizing the orders of the necessary directions, we can end up with a room layout like
22  * up up left up left up up
23  * We still get to the same room, but in a way that would feel much more organic.
24  * ----------
25  * Now that we have the order of directions we need to find rooms that can have doors through the corresponding walls. The starting
26  * room is free on all 4 sides, we would need a room above the first room that is not blocked in either the up or down direction.
27  * Then we need a room above this room that is not blocked on its left or bottom, etc...
28  * ----------
29  * After the "spine" of the floor has been built up, we add random rooms/branches to the existing rooms, expanding the map.
30  */

92  public List<Direction> DeltaPositionToDirections(int dX, int dY)
93  {
94  List<Direction> directions = new List<Direction>();
95 
96  while (Math.Abs(dX) + Math.Abs(dY) > 0)
97  {
98  // Randomly choosing a direction (with proportional probability)
99  if (this.random.Next(0, Math.Abs(dX) + Math.Abs(dY)) >= Math.Abs(dY))
100  {
101  // Chose horizontal (X) movement
102  if (dX > 0)
103  {
104  dX--;
105  directions.Add(Direction.Right);
106  }
107  else
108  {
109  dX++;
110  directions.Add(Direction.Left);
111  }
112  }
113  else
114  {
115  // Chose vertical (Y) movement
116  if (dY > 0)
117  {
118  dY--;
119  directions.Add(Direction.Down);
120  }
121  else
122  {
123  dY++;
124  directions.Add(Direction.Up);
125  }
126  }
127  }
128 
129  return directions;
130  }

132  private void CreateBranches()
133  {
134  // Number of branches.
135  int branches = this.random.Next(3, 3 + (this.difficulty * 2));
136 
137  for (int i = 0; i < branches; i++)
138  {
139  int branchLength = this.random.Next(1, (int)(3 + (this.difficulty * 1.2)));
140 
141  // The room we branch out from.
142  Room existingRoom;
143 
144  // The new branche's offset from the existing room.
145  Point offset;
146  Point totalOffset;
147 
148  // The directions the rooms are connected.
149  Queue<Direction> directions;
150 
151  do
152  {
153  offset = this.SplitDistanceIntoDirections(branchLength);
154 
155  // Getting a random room the branch connects to.
156  existingRoom = this.GetRandomExistingRoom();
157 
158  totalOffset = new Point(existingRoom.Offset.X + offset.X, existingRoom.Offset.Y + offset.Y);
159 
160  // Creating a direction sequencre that contains which direction the rooms are connected to eachother.
161  directions = new Queue<Direction>(this.DeltaPositionToDirections(offset.X, offset.Y));
162  }
163 
164  // If the selected existing room is not open to the first direction, or already has a neighbour to that direction, we try again
165  while (this.floor.GetRooms().Any(r => r.Offset == totalOffset) || !existingRoom.Layout.IsOpenToThe(directions.Peek()) || existingRoom.Neighbours.GetRoomToThe(directions.Peek()) != null);
166 
167  // Getting a layout that can be at the end of the branch.
168  Layout roomLayout = this.GetLayoutByFreeDirection(Utilities.ReverseDirection(directions.Last()));
169 
170  // Creating the instance of the room that will be at the end of the branch.
171  Room endingRoom = new Room(roomLayout, totalOffset);
172  this.ConnectToNeighbours(endingRoom);
173 
174  // Connecting the existing and the new room.
175  this.ConnectRooms(existingRoom, endingRoom, directions);
176  }
177  }

183  private void ConnectToNeighbours(Room newRoom)
184  {
185  // Loop through all neighbours.
186  foreach (Room room in this.floor.GetRooms())
187  {
188  bool isNeighbour = true;
189  Direction dir = default;
190  if (room.Offset.Y == newRoom.Offset.Y)
191  {
192  if (room.Offset.X == newRoom.Offset.X + 1)
193  {
194  dir = Direction.Right;
195  }
196  else if (room.Offset.X == newRoom.Offset.X - 1)
197  {
198  dir = Direction.Left;
199  }
200  else
201  {
202  isNeighbour = false;
203  }
204  }
205  else if (room.Offset.X == newRoom.Offset.X)
206  {
207  if (room.Offset.Y == newRoom.Offset.Y + 1)
208  {
209  dir = Direction.Down;
210  }
211  else if (room.Offset.Y == newRoom.Offset.Y - 1)
212  {
213  dir = Direction.Up;
214  }
215  else
216  {
217  isNeighbour = false;
218  }
219  }
220  else
221  {
222  isNeighbour = false;
223  }
224 
225  // If they are in fact naighbours.
226  if (isNeighbour)
227  {
228  // Checking if the layouts are clear
229  if (newRoom.Layout.IsOpenToThe(dir) && room.Layout.IsOpenToThe(Utilities.ReverseDirection(dir)))
230  {
231  newRoom.Neighbours.SetRoomToThe(dir, room);
232  room.Neighbours.SetRoomToThe(Utilities.ReverseDirection(dir), newRoom);
233  }
234  }
235  }
236  }

247  private void CreateStartExitConnection()
248  {
249  // Placing the starting room.
250  this.floor.StartingRoom = this.GetStartingRoomInstance();
251  this.floor.CurrentRoom = this.floor.StartingRoom;
252 
253  // Randomizing the distance of the exit room from the starting room (respecting difficulty)
254  int distance = 4 + Convert.ToInt32((this.difficulty + this.random.Next(0, 2)) * 1.2);
255 
256  // Splitting the distance (= offset) between X and Y coordinates.
257  Point offsets;
258 
259  do
260  {
261  offsets = this.SplitDistanceIntoDirections(distance);
262  }
263  while (offsets.X == 0 || offsets.Y == 0); // Making sure the walk is not a straight line
264 
265  Room exitRoom = this.GetExitRoomInstance(offsets);
266 
267  // Connecting the starting and ending rooms
268  this.ConnectRooms(this.floor.StartingRoom, exitRoom);
269  }

271  private Point SplitDistanceIntoDirections(int distance)
272  {
273  int xOffset = this.random.Next(-distance + 1, distance);
274  int yOffset = ((this.random.Next(0, 2) * 2) - 1) * (distance - Math.Abs(xOffset));
275  return new Point(xOffset, yOffset);
276  }

299  private void ConnectRooms(Room currentRoom, Room targetRoom, Queue<Direction> directions)
300  {
301  // Getting which direction we need to leave the current room.
302  Direction nextDirection = directions.Dequeue();
303 
304  // If this is the last recursion, we should be at the last room.
305  if (directions.Count == 0)
306  {
307  currentRoom.Neighbours.SetRoomToThe(nextDirection, targetRoom);
308  targetRoom.Neighbours.SetRoomToThe(Utilities.ReverseDirection(nextDirection), currentRoom);
309  return;
310  }
311 
312  Point offset = new Point(
313  currentRoom.Offset.X + Utilities.DirectionToXOffset(nextDirection),
314  currentRoom.Offset.Y + Utilities.DirectionToYOffset(nextDirection));
315 
316  // If there is already a room at that location, we don't want to overwrite it.
317  Room roomAtThatOffset = this.floor.GetRooms().FirstOrDefault(r => r.Offset == offset);
318 
319  if (roomAtThatOffset == default)
320  {
321  // No room at that location, we can create a new one.
322  roomAtThatOffset = new Room(
323  this.GetLayoutByFreeDirections(
324  Utilities.ReverseDirection(nextDirection),
325  directions.Peek()),
326  offset);
327  }
328 
329  // Setting the current and new room to be neighbours.
330  if (roomAtThatOffset.Layout.IsOpenToThe(Utilities.ReverseDirection(nextDirection)))
331  {
332  currentRoom.Neighbours.SetRoomToThe(nextDirection, roomAtThatOffset);
333  this.ConnectToNeighbours(roomAtThatOffset);
334  }
335 
336  // If we ran into an existing room, we need to check if that room has a layout that allows the branch to continue.
337  if (roomAtThatOffset.Layout.IsOpenToThe(directions.Peek()))
338  {
339  this.ConnectRooms(roomAtThatOffset, targetRoom, directions);
340  }
341  }

378  private Layout GetLayoutByFreeDirections(Direction freeDirection1, Direction freeDirection2)
379  {

381  .Where(

383  r.Type == RoomType.Normal &&
384  r.IsOpenToThe(freeDirection1) &&
