98  // Generating the random number of clutter textures on the ground.
99  int pieces = this.Random.Next(3, 15);
100 
101  // Keeps track of where clutter has been placed (to avoid overlaying textures)
102  List<Point> placedLocations = new List<Point>();
103  for (int i = 0; i < pieces; i++)
104  {
105  Point point = default;
106  string texture = clutterTextures[this.Random.Next(clutterTextures.Length)];
107  do
108  {
109  int randX = this.Random.Next(Layout.RoomWidth);
110  int randY = this.Random.Next(Layout.RoomHeight);
111 
112  point = new Point(randX, randY);
113  }
114  while (placedLocations.Any(p => p.Equals(point)));
115  placedLocations.Add(point);
116  group.Children.Add(this.CreateGridDrawing((int)point.X + 1, (int)(point.Y + 2), texture));
117  }
118 
119  return group;
