104  public void MoveToRoom(Room room, Direction startingDirection)

108  Player player = GameModel.GetInstance().Player;
109  Rect playerSize = GameModel.GetInstance().Player.Hitbox.RelativeHitbox;
110 
111  switch (startingDirection)

114  startingPoint = new PointF((float)-playerSize.Left + 0.01f, player.Location.Y);
115  break;
116 
117  case Direction.Right:
118  startingPoint = new PointF((float)(Layout.RoomWidth - playerSize.Width), player.Location.Y);
119  break;
120 
121  case Direction.Up:
122  startingPoint = new PointF(player.Location.X, (float)-playerSize.Top + 0.01f);
123  break;
124 
125  case Direction.Down:
126  startingPoint = new PointF(player.Location.X, (float)(Layout.RoomHeight - playerSize.Height - playerSize.Top - 0.01f));
127  break;
128  }
129 
130  this.MoveToRoom(room, startingPoint);
131  }
