64  private void UpdateHealth()
65  {
66  this.Healthbar = new DrawingGroup();
67 
68  double heartSize = this.TileSize / 1.4;
69 
70  // Calculating number of hearts
71  int fullHearts = Convert.ToInt32(Math.Floor(this.Model.Player.Health / 2));
72  int emptyHearts = Convert.ToInt32(Math.Floor((this.Model.Player.MaxHealth - this.Model.Player.Health) / 2));
73  int halfHeart = this.Model.Player.Health % 2 == 1 ? 1 : 0;
74 
75  // Drawing full hearts
76  for (int i = 0; i < fullHearts; i++)
77  {
78  this.Healthbar.Children.Add(
79  CreateGeoDrawing(
80  new Rect(this.GUIOffset + (i * heartSize), this.GUIOffset, heartSize, heartSize),
81  "ui_heart_full"));
82  }
83 
84  // Drawing half heart
85  if (halfHeart == 1)
86  {
87  this.Healthbar.Children.Add(
88  CreateGeoDrawing(
89  new Rect(this.GUIOffset + (fullHearts * heartSize), this.GUIOffset, heartSize, heartSize),
90  "ui_heart_half"));
91  }
92 
93  // Drawing empty hearts
94  for (int i = 0; i < emptyHearts; i++)
95  {
96  this.Healthbar.Children.Add(
97  CreateGeoDrawing(
98  new Rect(
99  this.GUIOffset + ((i + fullHearts + halfHeart) * heartSize), this.GUIOffset, heartSize, heartSize),
100  "ui_heart_empty"));
101  }
102 
103  FormattedText formattedText = new FormattedText(
104  "Score: " + this.Model.Player.Stats.Score,
105  CultureInfo.GetCultureInfo("en-us"),
106  FlowDirection.LeftToRight,
107  this.font.GetTypefaces().First(),
108  this.FontSize,
109  Brushes.Black);
110 
111  Geometry textGeo = formattedText.BuildGeometry(new Point(this.GUIOffset, this.GUIOffset + (heartSize * 2)));
112  this.Healthbar.Children.Add(new GeometryDrawing(new SolidColorBrush(Color.FromRgb(255, 255, 255)), null, textGeo));
113  }

115  private void UpdateMinimap()
116  {
117  this.Minimap = new DrawingGroup();
118 
119  this.Minimap.Children.Add(CreateGeoDrawing(this.MinimapRect, this.MinimapBackground));
120 
121  List<Room> rooms = this.Model.CurrentFloor.GetRooms()
122  .Where(r => r.IsVisibleOnMinimap)
123  .ToList();
124 
125  Size rectSize = new Size(
126  this.TileSize * Layout.RoomWidth * UIRoomScale,
127  this.TileSize * Layout.RoomHeight * UIRoomScale);
128 
129  Point currentRoomPos = new Point(
130  this.MinimapRect.Left + ((this.MinimapRect.Width - rectSize.Width) / 2),
131  this.MinimapRect.Top + ((this.MinimapRect.Height - rectSize.Height) / 2));
132 
133  // Adding rooms
134  foreach (Room room in rooms)
135  {
136  Point pos = new Point(
137  currentRoomPos.X + ((room.Offset.X - this.Model.CurrentFloor.CurrentRoom.Offset.X) * (rectSize.Width + 2)),
138  currentRoomPos.Y + ((room.Offset.Y - this.Model.CurrentFloor.CurrentRoom.Offset.Y) * (rectSize.Height + 2)));
139  Rect roomRect = new Rect(pos, rectSize);
140 
141  // Only draw on the minimap
142  roomRect.Intersect(this.MinimapRect);
143 
144  double cornerRadius = roomRect.Height / 8;
145  Geometry geo = new RectangleGeometry(roomRect, cornerRadius, cornerRadius);
146  this.Minimap.Children.Add(new GeometryDrawing(this.GetMinimapRoomBrush(room), null, geo));
147  }
148 
149  // Drawing border and background
150  this.Minimap.Children.Add(CreateGeoDrawing(this.MinimapRect, this.MinimapBorder));
151 
152  // Writing floor number
153  var tfc = this.font.GetTypefaces();
154 
155  FormattedText formattedText = new FormattedText(
156  "Floor " + this.Model.CurrentFloor.Difficulty,
157  CultureInfo.GetCultureInfo("en-us"),
158  FlowDirection.LeftToRight,
159  this.font.GetTypefaces().First(),
160  this.FontSize,
161  Brushes.Black);
162 
163  Geometry textGeometry = formattedText.BuildGeometry(
164  new Point(this.MinimapRect.Left + ((this.MinimapRect.Width - formattedText.Width) / 2), this.MinimapRect.Bottom + this.GUIOffset));
165 
166  this.Minimap.Children.Add(new GeometryDrawing(new SolidColorBrush(Color.FromRgb(255, 255, 255)), null, textGeometry));
167  }
