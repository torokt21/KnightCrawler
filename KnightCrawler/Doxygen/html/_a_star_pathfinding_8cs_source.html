58  public override void OnTick(double deltaTime)
59  {
60  // If the path is null, then there is no path to the player.
61  if (this.Path != null)
62  {
63  // If the path contains more than one items, the target's tile has been reached.
64  if (this.Path.Count > 0)
65  {
66  double dx = this.Path[0].Rectangle.Left + (this.Path[0].Rectangle.Width / 2) - this.Entity.Hitbox.Center.X;
67  double dy = this.Path[0].Rectangle.Top + (this.Path[0].Rectangle.Height / 2) - this.Entity.Hitbox.Center.Y;
68 
69  dx = (Math.Abs(dx) < SameLineThreshold) ? 0 : dx;
70  dy = (Math.Abs(dy) < SameLineThreshold) ? 0 : dy;
71 
72  if (!this.Entity.Hitbox.Rectangle.IntersectsWith(this.Target.Hitbox.Rectangle))
73  {
74  this.Entity.MoveDirectional(
75  Math.Sign(dx),
76  Math.Sign(dy),
77  (float)this.Speed,
78  deltaTime,
79  false);
80  }
81  }
82 
83  // If the path is 0 items long, the enemy has to mindlessly chase the player
84  else
85  {
86  this.ChasingBehaviour.OnTick(deltaTime);
87  }
88  }

97  private static int GetDistance(AStarNode node1, AStarNode node2)
98  {
99  int distX = (int)Math.Abs(node1.Tile.Rectangle.X - node2.Tile.Rectangle.X);
100  int distY = (int)Math.Abs(node1.Tile.Rectangle.Y - node2.Tile.Rectangle.Y);
101 
102  if (distX > distY)
103  {
104  return (WeightDiagonal * distY) + (WeightAcross * (distX - distY));
105  }
106  else
107  {
108  return (WeightDiagonal * distY) + (WeightAcross * (distY - distX));
109  }
110  }

115  private void FindPath()
116  {
117  // Turning tiles into nodes.
118  AStarNode[,] allNodes = new AStarNode[Layout.RoomWidth, Layout.RoomHeight];
119  for (int x = 0; x < Layout.RoomWidth; x++)
120  {
121  for (int y = 0; y < Layout.RoomHeight; y++)
122  {
123  allNodes[x, y] = new AStarNode(this.Entity.Room.Tiles[x, y]);
124  }
125  }
126 
127  // Getting start and target nodes.
128  FloorTile startTile = this.Entity.TileUnderCenter;
129  FloorTile targetTile = this.Target.TileUnderCenter;
130 
131  // Turning the target and start tiles into nodes
132  AStarNode startNode = allNodes[(int)startTile.Rectangle.X, (int)startTile.Rectangle.Y];
133  AStarNode targetNode = allNodes[(int)targetTile.Rectangle.X, (int)targetTile.Rectangle.Y];
134 
135  // The collection of nodes that are yet to be evaluated
136  List<AStarNode> openSet = new List<AStarNode>();
137 
138  // Nodes that are already evaluated
139  HashSet<AStarNode> closedSet = new HashSet<AStarNode>();
140 
141  openSet.Add(startNode);
142 
143  while (openSet.Count > 0)
144  {
145  AStarNode currentNode = openSet[0];
146 
147  // Finding the lowest FCost in the open set.
148  for (int i = 1; i < openSet.Count; i++)
149  {
150  // Finding the node with the lowest F cost (closest to the starting node)
151  if (openSet[i].FCost <= currentNode.FCost)
152  {
153  // If the F costs are the same, we compare H costs
154  if (openSet[i].FCost == currentNode.FCost && openSet[i].HCost < currentNode.HCost)
155  {
156  currentNode = openSet[i];
157  }
158  }
159  }
160 
161  // We store the current node as evaluated.
162  openSet.Remove(currentNode);
163  closedSet.Add(currentNode);
164 
165  // If the current node is the target node, we have our path.
166  if (currentNode == targetNode)
167  {
168  this.Path = this.RetracePath(startNode, targetNode);
169  return;
170  }
171 
172  // Looping through the current node's neughbours.
173  foreach (AStarNode neighbour in this.GetNeighbours(currentNode, allNodes))
174  {
175  // We ignore tiles that block pathfinding, or have already been evaluated.
176  if (neighbour.Tile.BlocksPathFinding || closedSet.Contains(neighbour))
177  {
178  continue;
179  }
180 
181  // Calculating the cost to this neighbor.
182  int newMovementCostToNeighbour = currentNode.GCost + GetDistance(currentNode, neighbour);
183 
184  // If the calculated cost is lower than the previous cost (or has never been evaluated)
185  if (newMovementCostToNeighbour < neighbour.GCost || !openSet.Contains(neighbour))
186  {
187  // We save the G cost
188  neighbour.GCost = newMovementCostToNeighbour;
189 
190  // Calculate H cost (distance from target)
191  neighbour.HCost = GetDistance(neighbour, targetNode);
192 
193  // Storing the current node as it's previous node
194  neighbour.ParentNode = currentNode;
195 
196  // Marking the node as evaluated.
197  if (!openSet.Contains(neighbour))
198  {
199  openSet.Add(neighbour);
200  }
201  }
202  }
203  }
204  }

226  private List<FloorTile> RetracePath(AStarNode startNode, AStarNode targetNode)
227  {
228  List<FloorTile> tiles = new List<FloorTile>();
229  AStarNode current = targetNode;
230 
231  while (current != startNode)
232  {
233  tiles.Add(current.Tile);
234  current = current.ParentNode;
235  }
236 
237  tiles.Reverse();
238  return tiles;

247  private List<AStarNode> GetNeighbours(AStarNode node, AStarNode[,] nodeMatrix)
248  {
249  List<AStarNode> tiles = new List<AStarNode>();
250  for (int x = -1; x <= 1; x++)
251  {
252  for (int y = -1; y <= 1; y++)
253  {
254  // Ignoring center (it's the current node)
255  if (x == 0 && y == 0)
256  {
257  continue;
258  }
259 
260  // The actual x and y coodrinates to check
261  int checkX = (int)node.Tile.Rectangle.X + x;
262  int checkY = (int)node.Tile.Rectangle.Y + y;
263 
264  // Checking against room tile edges
265  if (checkX >= 0 && checkX < Layout.RoomWidth && checkY >= 0 && checkY < Layout.RoomHeight)
266  {
267  // Checking if the diagonal movement would be stopped by any diagonal obstacles.
268  if (!(Math.Abs(x) == 1 &&
269  Math.Abs(y) == 1 &&
270  (nodeMatrix[checkX, (int)node.Tile.Rectangle.Y].Tile.BlocksPathFinding ||
271  nodeMatrix[(int)node.Tile.Rectangle.X, checkY].Tile.BlocksPathFinding)))
272  {
273  tiles.Add(nodeMatrix[checkX, checkY]);
274  }
275  }
276  }
277  }
278 
279  return tiles;
280  }
