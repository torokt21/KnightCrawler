141  public List<FloorTile> TilesNear(Rect rectangle, float radius = 1)
142  {
143  List<FloorTile> tiles = new List<FloorTile>();
144 
145  int xMin = Convert.ToInt32(Math.Max(0, Math.Floor(rectangle.Left) - radius));
146  int xMax = Convert.ToInt32(Math.Min(Math.Ceiling(rectangle.Right) + radius, Layout.RoomWidth));
147 
148  int yMin = Convert.ToInt32(Math.Max(0, Math.Floor(rectangle.Top) - radius));
149  int yMax = Convert.ToInt32(Math.Min(Math.Ceiling(rectangle.Bottom) + radius, Layout.RoomHeight));
150 
151  for (int y = yMin; y < yMax; y++)
152  {
153  for (int x = xMin; x < xMax; x++)
154  {
155  tiles.Add(this.Tiles[x, y]);
156  }
157  }
158
