147  public void MoveDirectional(int dx, int dy, float speed, double deltaTime, bool hitTest)
148  {
149  float realSpeed = SpeedFromDirection(dx, dy, speed) * (float)deltaTime;
150 
151  // Calculating the target position
152  float newPosX = this.Location.X + (dx * realSpeed);
153  float newPosY = this.Location.Y + (dy * realSpeed);
154 
155  Room room = this.Room;
156  this.DoWallDetection(ref newPosX, ref newPosY);
157 
158  if (this.Room != room)
159  {
160  return;
161  }
162 
163  if (hitTest)
164  {
165  Rect hitboxRect = new Rect(newPosX + this.Hitbox.RelativeHitbox.Left, newPosY + this.Hitbox.RelativeHitbox.Top, this.Hitbox.RelativeHitbox.Width, this.Hitbox.RelativeHitbox.Height);
166 
167  // Calculating the new positions's hitbox center, to find nearby obstacles.
168  PointF centerPoint = new PointF(
169  (float)(hitboxRect.Left + (hitboxRect.Width / 2)),
170  (float)(hitboxRect.Top + (hitboxRect.Height / 2)));
171 
172  // Hittesting with nearby obstacles
173  if (this.Room.TilesNear(this.Hitbox.Rectangle).Any(t => t.BlocksEntities && hitboxRect.IntersectsWith(t.Rectangle)))
174  {
175  // If moving diagonally, we split the movement into two slower movements.
176  if (dx != 0 && dy != 0)
177  {
178  this.MoveDirectional(dx, 0, realSpeed, 1, true);
179  this.MoveDirectional(0, dy, realSpeed, 1, true);
180  }
181 
182  return;
183  }
184  }
