117  protected override void DoWallDetection(ref float newPosX, ref float newPosY)

121  if (this.Room.IsCleared)

123  // Detection if the player is leaving the room.
124  if (this.IsCenteredHorizontally(newPosX))
125  {
126  // The player is centered horizontally and going up.
127  if (newPosY + this.Hitbox.RelativeHitbox.Top < 0 && this.Room.Neighbours.GetRoomToThe(Direction.Up) != null)
128  {
129  (this as Player).OnLeavingRoom(this.Room, Direction.Up);
130  roomLeft = true;
131  }
132 
133  // The player is centered horizontally and going down.
134  else if (newPosY > Layout.RoomHeight - this.Size.Height && this.Room.Neighbours.GetRoomToThe(Direction.Down) != null)
135  {
136  (this as Player).OnLeavingRoom(this.Room, Direction.Down);
137  roomLeft = true;
138  }
139  }
140 
141  // Checking if the player is centered vertically
142  if (this.IsCenteredVertically(newPosY))
143  {
144  // The player is centered vertically and moving to the right.
145  if (newPosX < 0 && this.Room.Neighbours.GetRoomToThe(Direction.Left) != null)
146  {
147  (this as Player).OnLeavingRoom(this.Room, Direction.Left);
148  roomLeft = true;
149  }
150 
151  // The player is centered vertically and moving to the left.
152  else if (newPosX > Layout.RoomWidth - this.Hitbox.Rectangle.Width && this.Room.Neighbours.GetRoomToThe(Direction.Right) != null)
153  {
154  (this as Player).OnLeavingRoom(this.Room, Direction.Right);
155  roomLeft = true;
