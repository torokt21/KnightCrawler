202  private void RoomRenderer_SizeChanged(object sender, SizeChangedEventArgs e)
203  {
204  // Calculating the dimensions of one cell.
205  this.TileSize = (int)Math.Ceiling(this.ActualHeight / (Layout.RoomHeight + 3));
206 
207  // The left offset of all drawings, to keep the room centered.
208  this.LeftOffset = Convert.ToInt32((this.ActualWidth - ((Layout.RoomWidth + 2) * this.TileSize)) / 2);
209 
210  // Updating the minimap rectangle
211  Size minimapSize = new Size(this.TileSize * 3, this.TileSize * 2);
212  Point minimapPosition = new Point(this.ActualWidth - this.GUIOffset - minimapSize.Width, this.GUIOffset);
213 
214  this.MinimapRect = new Rect(minimapPosition, minimapSize);
215 
216  // The room size has changed, so the saved drawings need to be redrawn.
217  this.UpdateRoom();
218  }
